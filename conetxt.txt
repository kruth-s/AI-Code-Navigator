import os
import re

BASE_PATH = "markdown_rules"

def load_index():
    with open(os.path.join(BASE_PATH, "MARKDOWN_INDEX.md"), "r") as f:
        content = f.read()

    sections = re.split(r"## ", content)[1:]
    parsed = []

    for sec in sections:
        lines = sec.strip().split("\n")
        title = lines[0]
        file = [l for l in lines if l.startswith("file:")][0].split(":")[1].strip()
        keywords = [l for l in lines if l.startswith("keywords:")][0].split(":")[1].strip().split(", ")

        parsed.append({
            "title": title,
            "file": file,
            "keywords": keywords
        })

    return parsed
🔍 Step 2: Smart Section Selection (WITHOUT embeddings yet)
Instead of jumping to vector DB, do this:

def find_relevant_sections(code, index_data):
    relevant = []

    for section in index_data:
        for keyword in section["keywords"]:
            if keyword.lower() in code.lower():
                relevant.append(section)
                break

    return relevant
👉 This acts as your “cheap semantic router”

📄 Step 3: Load Only Needed Markdown
def load_rules(sections):
    rules = []

    for sec in sections:
        path = os.path.join(BASE_PATH, sec["file"])
        with open(path, "r") as f:
            rules.append({
                "section": sec["title"],
                "content": f.read()
            })

    return rules
🧠 Step 4: LLM Prompt
def build_prompt(code, rules):
    rules_text = "\n\n".join([r["content"] for r in rules])

    return f"""

{rules_text}

{code}

- Violations
- Fixes
- Severity
